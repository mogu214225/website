public class DataAnalysis

      double iddev = 0, Gradedev = 0, Scoredev = 0, Absencedev = 0, Timedev = 0,
      Startdev = 0, Subdev = 0, Quizdev = 0, Orderdev = 0;

      s.useDelimiter("\t");
      while (s.hasNext())
      {
         n++;
         id = s.next().trim();
         Grade = s.next().trim();
         Score = s.next().trim();
         Absence = s.next().trim();
         Time = s.next().trim();
         Start = s.next().trim();
         Sub = s.next().trim();
         Quiz = s.next().trim();
         s.useDelimiter("\n");
         Order = s.next().trim();
         s.useDelimiter("\t");
         
         storestring.append(id).append("\t");
         storestring.append(Grade).append("\t");
         storestring.append(Score).append("\t");
         storestring.append(Absence).append("\t");
         storestring.append(Time).append("\t");
         storestring.append(Start).append("\t");
         storestring.append(Sub).append("\t");
         storestring.append(Quiz).append("\t");
         storestring.append(Order).append("\n");
         
         if (n == 1) continue; // ignore the heading
            
         idsum += Double.parseDouble(id);
         Scoresum += Double.parseDouble(Score);
         Absencesum += Double.parseDouble(Absence);
         Timesum += Double.parseDouble(Time);
         Startsum += Double.parseDouble(Start);
         Subsum += Double.parseDouble(Sub);
         Quizsum += Double.parseDouble(Quiz);
         Ordersum += Double.parseDouble(Order);
         
         if (Grade.equals("A")) GradeNum = 4;
            else if (Grade.equals("B")) GradeNum = 3;
               else if (Grade.equals("C")) GradeNum = 2;
                  else if (Grade.equals("D")) GradeNum = 1;
                     else GradeNum = 0;
                        Gradesum += GradeNum;
      }
      
      n--; // Subtract 1 to exclude the header row
      idmean = idsum / n;
      Grademean = Gradesum / n;
      Scoremean = Scoresum / n;
      Absencemean = Absencesum / n;
      Timemean = Timesum / n;
      Startmean = Startsum / n;
      Submean = Subsum / n;
      Quizmean = Quizsum / n;
      Ordermean = Ordersum / n;

      System.out.printf("Scoremean: %.2f%n", Scoremean);
      System.out.printf("Absencemean: %.2f%n", Absencemean);
      System.out.printf("Timemean: %.2f%n", Timemean);
      System.out.printf("Startmean: %.2f%n", Startmean);
      System.out.printf("Submean: %.2f%n", Submean);
      System.out.printf("Quizmean: %.2f%n", Quizmean);
      System.out.printf("Ordermean: %.2f%n", Ordermean);
      
      s = new Scanner(storestring.toString().trim());
      idsum = Gradesum = Scoresum = Absencesum = Timesum = Startsum = Subsum = Quizsum = Ordersum = 0;
      s.useDelimiter("\t");

      {
         id = s.next().trim();
         Grade = s.next().trim();
         Score = s.next().trim();
         Absence = s.next().trim();
         Time = s.next().trim();
         Start = s.next().trim();
         Sub = s.next().trim();
         Quiz = s.next().trim();
         s.useDelimiter("\n");
         Order = s.next().trim();
         s.useDelimiter("\t");
         
         if (id.equals("id")) continue;
            
         idsum += Math.pow(Double.parseDouble(id) - idmean, 2);
         Scoresum += Math.pow(Double.parseDouble(Score) - Scoremean, 2);
         Absencesum += Math.pow(Double.parseDouble(Absence) - Absencemean, 2);

            else if (Grade.equals("B")) GradeNum = 3;
               else if (Grade.equals("C")) GradeNum = 2;
                  else if (Grade.equals("D")) GradeNum = 1;
                     else GradeNum = 0;
                        Gradesum += Math.pow(GradeNum - Grademean, 2);
      }
      
      iddev = Math.sqrt(idsum / n);
      Gradedev = Math.sqrt(Gradesum / n);
      Scoredev = Math.sqrt(Scoresum / n);
      Absencedev = Math.sqrt(Absencesum / n);
      Timedev = Math.sqrt(Timesum / n);
      Startdev = Math.sqrt(Startsum / n);
      Subdev = Math.sqrt(Subsum / n);
      Quizdev = Math.sqrt(Quizsum / n);
      Orderdev = Math.sqrt(Ordersum / n);
      
      System.out.printf("iddev: %.2f%n", iddev);
      System.out.printf("Gradedev: %.2f%n", Gradedev);
      System.out.printf("Scoredev: %.2f%n", Scoredev);
      System.out.printf("Absencedev: %.2f%n", Absencedev);
      System.out.printf("Timedev: %.2f%n", Timedev);
      System.out.printf("Startdev: %.2f%n", Startdev);
      System.out.printf("Subdev: %.2f%n", Subdev);
      System.out.printf("Quizdev: %.2f%n", Quizdev);
      System.out.printf("Orderdev: %.2f%n", Orderdev);
      
      s.close();
   }

import java.awt.Color;
import java.awt.Graphics;
import javax.swing.JFrame;

public class TurkishFlag extends FlagCanvas
{
   private int flagWidth;
   private int flagHeight;
   
   public TurkishFlag(int w)
   {
      super(w);
      flagWidth = w;
      flagHeight = w * 2 / 3;
   }
   
   void paintMoon(Graphics g, java.awt.Color bgcolor, int radius, int centerx, int centery)
   {
      g.setColor(bgcolor);
      g.fillOval(centerx - radius, centery - radius, radius * 2, radius * 2);
   }
   
   public void paintStar(double centerX, double centerY, double radius, double ang, Graphics g)
   {
      g.setColor(Color.WHITE); // white star
      int[][] d = vertices(centerX, centerY, radius, ang);
      g.fillPolygon(d[0], d[1], 10);
   }
   
   public void paint(Graphics g)
   {
      g.setColor(Color.RED);
      g.fillRect(0, 0, flagWidth, flagHeight);
      
      // Calculate dimensions based on given ratios
      int G = flagWidth;
      int A = G / 2;
      int B = G / 2;
      int C = (int)(0.0625 * G);
      int D = (int)(0.4 * G);
      int E = G / 3;
      int F = G / 4;
      
      // Calculate centers and radii
      int outerMoonCenterX = A;
      int outerMoonCenterY = flagHeight / 2;
      int outerMoonRadius = B / 2;
      
      int innerMoonCenterX = outerMoonCenterX + C;
      int innerMoonCenterY = outerMoonCenterY;
      int innerMoonRadius = D / 2;
      
      int starCenterX = outerMoonCenterX + E;
      int starCenterY = outerMoonCenterY;
      int starRadius = F / 2;
      
      // Paint outer moon (white)
      paintMoon(g, Color.WHITE, outerMoonRadius, outerMoonCenterX, outerMoonCenterY);
      
      // Paint inner moon (red, to create crescent shape)
      paintMoon(g, Color.RED, innerMoonRadius, innerMoonCenterX, innerMoonCenterY);
      
      // Paint star
      paintStar(starCenterX, starCenterY, starRadius, 0, g);
   }
   
   public static void main(String[] args)
   {
      JFrame frame = new JFrame("Turkish National Flag Simulation");
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      TurkishFlag canvas = new TurkishFlag(576);
      frame.setSize(canvas.flagWidth, canvas.flagHeight);
      frame.getContentPane().add(canvas);
      frame.setVisible(true);
   }
}

   /**

        */
   private void task()
   {
      try
      {
         long n = Long.parseLong(jTextField1.getText());
         int k = 0; // Counter for formatting output
         
         for (int i = 2; i <= Math.sqrt(n) + 1; i++)
         {
            if (done.get()) break; // Stop task if "Quit" is clicked
               
            if (n % i == 0)
            {
               k++;
               jTextArea1.append(i + (k % 10 == 0 ? "\n": ", "));
            }
         }
         
         if (!done.get())
         {
            jTextArea1.append("\nTask completed.\n");
         }
      }
      catch (NumberFormatException e)
      {
         jTextArea1.append("Invalid input. Please enter a valid number.\n");
      }

         jTextArea1.append("Factoring started...\n");
      }
      else
      {
         // Stop the factoring task
         done.set(true);
         jButton1.setText("Start");
         jTextArea1.append("Factoring stopped.\n");
      }
   }

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.IOException;
import java.net.*;
public class UDPChatApp extends JFrame
{
   private JTextArea sendTextArea;
   private JTextArea receiveTextArea;
   private DatagramSocket socket;
   private InetAddress receiverAddress;
   private int port = 2311;
   public UDPChatApp()
   {
      setTitle("UDP Chat Application");
      setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      setSize(600, 400);
      setLayout(new BorderLayout());
      sendTextArea = new JTextArea(5, 20);
      receiveTextArea = new JTextArea(15, 20);
      receiveTextArea.setEditable(false);
      JButton sendButton = new JButton("Send");
      sendButton.addActionListener(new ActionListener()
      {
         @Override
         public void actionPerformed(ActionEvent e)
         {
            sendMessage();
         }
      }
      );
      JPanel inputPanel = new JPanel();
      inputPanel.setLayout(new BorderLayout());
      inputPanel.add(new JScrollPane(sendTextArea), BorderLayout.CENTER);
      inputPanel.add(sendButton, BorderLayout.EAST);
      add(new JScrollPane(receiveTextArea), BorderLayout.CENTER);
      add(inputPanel, BorderLayout.SOUTH);
      setVisible(true);
      try
      {
         socket = new DatagramSocket(port);
         startListening();
      }
      catch (SocketException e)
      {
         e.printStackTrace();
      }
   }
   private void sendMessage()
   {
      String message = sendTextArea.getText();
      if (!message.isEmpty())
      {
         try
         {
            byte[] sendBuffer = message.getBytes();
            DatagramPacket sendPacket = new DatagramPacket(sendBuffer, sendBuffer.length, receiverAddress, port);
            socket.send(sendPacket);
            sendTextArea.setText("");
         }
         catch (IOException e)
         {
            e.printStackTrace();
         }
      }
   }
   private void startListening()
   {
      Thread receiveThread = new Thread(() ->
      {
         while (true)
         {
            try
            {
               byte[] receiveBuffer = new byte[1024];
               DatagramPacket receivePacket = new DatagramPacket(receiveBuffer, receiveBuffer.length);
               socket.receive(receivePacket);
               String receivedMessage = new String(receivePacket.getData(), 0, receivePacket.getLength());
               receiveTextArea.append(receivedMessage + "\n");
            }
            catch (IOException e)
            {
               e.printStackTrace();
            }
         }
      }
      );
      receiveThread.start();
   }
   public static void main(String[] args)
   {
      SwingUtilities.invokeLater(() ->
      {
         UDPChatApp app = new UDPChatApp();
         try
         {
            app.receiverAddress = InetAddress.getByName(JOptionPane.showInputDialog("Enter receiver's IP address"));
         }
         catch (UnknownHostException e)
         {
            e.printStackTrace();
         }
      }
      );
   }
}

         +"Content-Type: text/plain\r\n"
         +"\r\n"
         +"request received";
         out.writeBytes(response);
         in.close();
         out.close();
         clientSocket.close();
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }

   {
      Map headers = new HashMap<>();
      String line;
      while(!(line = in.readLine()).isEmpty())
      {
         String[] parts = line.split(":", 2);
         if(parts.length ==2)
         {
            headers.put(parts[0].trim(), parts[1].trim());
         }
      }
      return headers;
   }
